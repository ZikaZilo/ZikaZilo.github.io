package org.example.project.data_structure

// Assert One line is a Node and the csv file is a linked list.
// Assert the first line is the head of the linked list.
// Here are the functions to manipulate the linked list.

import java.io.File

fun LinkedList_clearNodes(link_list_file_path: String){
    if (!File(link_list_file_path).exists()){
        File(link_list_file_path).createNewFile()
    }
    File(link_list_file_path).writeText("")
}

fun LinkedList_appendNode(link_list_file_path: String, data: String){
    File(link_list_file_path).appendText("$data\n")
}

fun LinkedList_pushData(link_list_file_path: String, data: String){
    var original_data = File(link_list_file_path).readText()
    File(link_list_file_path).writeText("$data\n" + original_data)
}

fun LinkedList_printData(link_list_file_path: String){
    println(File(link_list_file_path).readText())
}

fun LinkedList_insertData(link_list_file_path: String, position: Int, value: String){
    var original_data = File(link_list_file_path).readText().split("\n")
    var final_data = mutableListOf<String>()
    // If the position is out of the List range, print error message
    if(original_data.size > position) {
        for (i in 0 until original_data.size) {
            if (i == position) {
                final_data.add(value)
            }
            final_data.add(original_data[i])
        }
        File(link_list_file_path).writeText(final_data.joinToString("\n"))
    }else{
        println("Position is out of range")
    }
}

fun LinedList_ifContains(link_list_file_path: String, value: String): Boolean{
    var original_data = File(link_list_file_path).readText().split("\n")
    for (i in 0 until original_data.size){
        if (original_data[i] == value){
            return true
        }
    }
    return false
}



fun LinkedList_popData(link_list_file_path: String){
    var original_data = File(link_list_file_path).readText().split("\n")
    var final_data = mutableListOf<String>()
    if(original_data.size > 0) {
        for (i in 1 until original_data.size) {
            final_data.add(original_data[i])
        }
        File(link_list_file_path).writeText(final_data.joinToString("\n"))
    }else{
        println("No data to pop")
    }
}

fun LinkedList_removeLastData(link_list_file_path: String){
    var original_data = File(link_list_file_path).readText().split("\n")
    var final_data = mutableListOf<String>()
    if(original_data.size > 0) {
        for (i in 0 until original_data.size - 1) {
            final_data.add(original_data[i])
        }
        File(link_list_file_path).writeText(final_data.joinToString("\n"))
    }else{
        println("No data to remove")
    }
}

fun LinkedList_removeDataInPosition(link_list_file_path: String, position: Int){
    var original_data = File(link_list_file_path).readText().split("\n")
    var final_data = mutableListOf<String>()
    if(original_data.size > position) {
        for (i in 0 until original_data.size) {
            if (i != position) {
                final_data.add(original_data[i])
            }
        }
        File(link_list_file_path).writeText(final_data.joinToString("\n"))
    }else{
        println("Position is out of range")
    }
}

fun LinkedList_getHead(link_list_file_path: String): String{
    return File(link_list_file_path).readText().split("\n")[0]
}

fun LinkedList_getTail(link_list_file_path: String): String{
    return File(link_list_file_path).readText().split("\n").last()
}

fun LinkedList_getDataInPosition(link_list_file_path: String, position: Int): String{
    return File(link_list_file_path).readText().split("\n")[position]
}

fun LinkedList_getLength(link_list_file_path: String): Int{
    return File(link_list_file_path).readText().split("\n").size
}

fun LinkedList_getAllData(link_list_file_path: String): List<String>{
    return File(link_list_file_path).readText().split("\n")
}


fun test_LinkedList(){
    var link_list_file_path = "./link_list.csv"

    // Test For clearNodes and appendNode
    LinkedList_clearNodes(link_list_file_path)
    LinkedList_appendNode(link_list_file_path,"1")
    LinkedList_appendNode(link_list_file_path, "2")

    // Test For insertData and removeDataInPosition
    println("After appending 1 and 2")
    LinkedList_printData(link_list_file_path)
    LinkedList_insertData(link_list_file_path, 1, "3")
    println("After inserting 3 at position 1")
    LinkedList_printData(link_list_file_path)
    LinkedList_insertData(link_list_file_path, 0, "4")
    println("After inserting 4 at position 0")
    LinkedList_removeDataInPosition(link_list_file_path, 1)
    println("After removing data at position 1")
    LinkedList_printData(link_list_file_path)


    // Test for pushData and popData
    LinkedList_pushData(link_list_file_path, "5")
    println("After pushing 5")
    LinkedList_printData(link_list_file_path)
    LinkedList_popData(link_list_file_path)
    println("After popping")
    LinkedList_printData(link_list_file_path)

    LinkedList_printData(link_list_file_path)

    // Test for removeLastData
    LinkedList_removeLastData(link_list_file_path)
    println("After removing last data")
    LinkedList_printData(link_list_file_path)

    // Test for getHead, getTail, getDataInPosition, getLength, getAllData
    LinkedList_getHead(link_list_file_path)
    println("Head: ${LinkedList_getHead(link_list_file_path)}")
    LinkedList_getTail(link_list_file_path)
    println("Tail: ${LinkedList_getTail(link_list_file_path)}")
    LinkedList_getDataInPosition(link_list_file_path, 0)
    println("Data at position 0: ${LinkedList_getDataInPosition(link_list_file_path, 0)}")
    LinkedList_getLength(link_list_file_path)
    println("Length: ${LinkedList_getLength(link_list_file_path)}")
    LinkedList_getAllData(link_list_file_path)
    println("All Data: ${LinkedList_getAllData(link_list_file_path)}")

    // Test for ifContains
    println("If contains 1: ${LinedList_ifContains(link_list_file_path, "1")}")
}

fun main(){
    test_LinkedList()
}
