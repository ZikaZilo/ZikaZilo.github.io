Python Tkinter Text Widget 详解 (◕‿◕✿)

Text Widget 是 Tkinter 中用于显示和编辑多行文本的强大组件，下面我会详细介绍它的使用方法和变量绑定技巧~

from tkinter import *

root = Tk()

# 创建Text Widget
text_box = Text(root, height=10, width=40)
text_box.pack()

root.mainloop()

常用配置选项:

  • height: 行数
  • width: 字符宽度
  • font: 字体设置
  • bg/fg: 背景/前景色
  • wrap: 换行方式 (WORD/CHAR/NONE)

def update_text(*args):
    text_box.delete(1.0, END)
    text_box.insert(END, text_var.get())

def sync_var(*args):
    text_var.set(text_box.get(1.0, END))

text_var = StringVar()
text_var.trace("w", update_text)  # 当变量变化时更新Text

text_box = Text(root, height=5, width=40)
text_box.pack()
text_box.bind("<KeyRelease>", sync_var)  # 当Text变化时更新变量

class TextVar:
    def __init__(self, text_widget=None):
        self._text_widget = text_widget
        self._callbacks = []
      
    def set(self, value):
        if self._text_widget:
            self._text_widget.delete(1.0, END)
            self._text_widget.insert(END, value)
        for callback in self._callbacks:
            callback()
          
    def get(self):
        return self._text_widget.get(1.0, END) if self._text_widget else ""
  
    def trace_add(self, mode, callback):
        if mode == "write":
            self._callbacks.append(callback)

# 使用示例
text_box = Text(root)
text_var = TextVar(text_box)

def on_change():
    print("Text changed:", text_var.get())

text_var.trace_add("write", on_change)
text_var.set("Initial value")

def find_text():
    search_str = search_entry.get()
    text_box.tag_remove("found", "1.0", END)
    idx = "1.0"
    while True:
        idx = text_box.search(search_str, idx, stopindex=END)
        if not idx: break
        last_idx = f"{idx}+{len(search_str)}c"
        text_box.tag_add("found", idx, last_idx)
        idx = last_idx
    text_box.tag_config("found", foreground="red")
